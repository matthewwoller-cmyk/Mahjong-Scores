            players.forEach(p => { if(p.isExpanded === undefined) p.isExpanded = false; });

        }


        window.onload = function() {

            if (isFirstLaunch) {

                showNameModal();

            }

        };


        function showNameModal() {

            const t = i18n[currentLang];

            document.getElementById('modal-title').innerText = t.modalTitle;

            document.getElementById('p1-name').placeholder = t.player + " 1";

            document.getElementById('p2-name').placeholder = t.player + " 2";

            document.getElementById('p3-name').placeholder = t.player + " 3";

            document.getElementById('modal-btn').innerText = t.modalBtn;


            // Pre-fill if they already have custom names (from a reset)

            document.getElementById('p1-name').value = players[0].name.startsWith("Player") || players[0].name.startsWith("玩家") ? "" : players[0].name;

            document.getElementById('p2-name').value = players[1].name.startsWith("Player") || players[1].name.startsWith("玩家") ? "" : players[1].name;

            document.getElementById('p3-name').value = players[2].name.startsWith("Player") || players[2].name.startsWith("玩家") ? "" : players[2].name;


            document.getElementById('name-modal').style.display = 'flex';

        }


        function saveInitialNames() {

            const t = i18n[currentLang];

            const p1 = document.getElementById('p1-name').value.trim();

            const p2 = document.getElementById('p2-name').value.trim();

            const p3 = document.getElementById('p3-name').value.trim();


            players[0].name = p1 || (t.player + " 1");

            players[1].name = p2 || (t.player + " 2");

            players[2].name = p3 || (t.player + " 3");


            document.getElementById('name-modal').style.display = 'none';

            saveData();

            render();

        }


        function saveData() {

            localStorage.setItem('mahjongData', JSON.stringify({ players, tableOffset }));

        }


        function saveStateForUndo() {

            historyStack.push({

                players: JSON.parse(JSON.stringify(players)),

                tableOffset: tableOffset

            });

            if (historyStack.length > 20) {

                historyStack.shift();

            }

        }


        function undoLastAction() {

            if (historyStack.length > 0) {

                const previousState = historyStack.pop(); 

                players = previousState.players;

                tableOffset = previousState.tableOffset;

                saveData();

                render();

            }

        }


        function applyManualAdjustment(isAdd) {

            const amountInput = document.getElementById('adjust-amount');

            let amount = parseInt(amountInput.value);

            

            if (isNaN(amount) || amount <= 0) return;

            

            saveStateForUndo();

            if (!isAdd) amount = -amount;


            tableOffset += amount;

            

            amountInput.value = '';

            saveData();

            render();

        }


        function toggleCollapse(index) {

            players[index].isExpanded = !players[index].isExpanded;

            saveData(); 

            render();

        }


        function halveScore(index) {

            saveStateForUndo(); 

            let p = players[index];

            

            if (p.score > 0) {

                p.score = Math.floor(p.score / 2);

            } else if (p.score < 0) {

                p.score = Math.ceil(p.score / 2);

            }

            

            p.runningTally += (p.score + p.setAsidePoints);

            

            p.score = 0;

            p.streakType = null;

            p.streakCount = 0;

            p.setAsidePoints = 0;

            

            saveData();

            render();

        }


        function render() {

            const t = i18n[currentLang];


            // Update Language Toggle Buttons

            document.getElementById('lang-en').classList.toggle('active', currentLang === 'en');

            document.getElementById('lang-zh').classList.toggle('active', currentLang === 'zh');


            // Update Static Text

            document.getElementById('ui-title').innerText = t.title;

            document.getElementById('ui-subtitle').innerText = t.subtitle;

            document.getElementById('ui-rules').innerText = t.rulesLink;

            document.getElementById('btn-undo').innerText = t.undoBtn;

            document.getElementById('btn-hard-reset').innerText = t.resetBtn;


            // Update Modal text dynamically in case language switches while it's open

            document.getElementById('modal-title').innerText = t.modalTitle;

            document.getElementById('p1-name').placeholder = t.player + " 1";

            document.getElementById('p2-name').placeholder = t.player + " 2";

            document.getElementById('p3-name').placeholder = t.player + " 3";

            document.getElementById('modal-btn').innerText = t.modalBtn;


            const board = document.getElementById('board');

            const scoreboard = document.getElementById('master-scoreboard');

            const manualAdjustContainer = document.getElementById('manual-adjust-container');

            const undoBtn = document.getElementById('btn-undo');

            

            board.innerHTML = ''; 

            undoBtn.disabled = (historyStack.length === 0);


            let globalNetScore = tableOffset;

            let playerTalliesHtml = '<div class="tally-row">';


            players.forEach((p, index) => {

                globalNetScore += p.runningTally;


                let tallyClass = 'neutral';

                let displayTally = p.runningTally;

                if (p.runningTally > 0) {

                    tallyClass = 'positive';

                    displayTally = "+" + p.runningTally;

                } else if (p.runningTally < 0) {

                    tallyClass = 'negative';

                }


                playerTalliesHtml += `

                    <div class="tally-item">

                        <div class="tally-name">${p.name}</div>

                        <div class="tally-value ${tallyClass}">${displayTally}</div>

                    </div>

                `;


                // --- Build the Player Cards ---

                let scoreClass = 'neutral';

                let displayScore = p.score;

                if (p.score > 0) {

                    scoreClass = 'positive';

                    displayScore = "+" + p.score;

                } else if (p.score < 0) {

                    scoreClass = 'negative';

                }


                let headerStreakText = '';

                if (p.streakCount > 0) {

                    const typeText = p.streakType === 'win' ? t.winStreak : t.lossStreak;

                    const color = p.streakType === 'win' ? "#27ae60" : "#e74c3c";

                    headerStreakText = `<span style="color: ${color}; font-weight: bold;">${p.streakCount} ${typeText}</span>`;

                }


                let headerBadgeHtml = '';

                if (p.setAsidePoints < 0) {

                    headerBadgeHtml = `<div class="header-badge bg-debt">${t.badgeDebt}${p.setAsidePoints}</div>`;

                } else if (p.setAsidePoints > 0) {

                    headerBadgeHtml = `<div class="header-badge bg-banked">${t.badgeBanked}${p.setAsidePoints}</div>`;

                }


                let expandedStreakHtml = `<div class="streak">${t.noStreak}</div>`;

                if (p.streakCount > 0) {

                    const typeText = p.streakType === 'win' ? t.winStreak : t.lossStreak;

                    const color = p.streakType === 'win' ? "#27ae60" : "#e74c3c";

                    expandedStreakHtml = `<div class="streak" style="color: ${color}">${p.streakCount} ${typeText}<br>(${t.multiplier})</div>`;

                }


                let isSpecialEnabled = (p.streakCount >= 3);

                let specialBtnClass = 'btn-special';

                let specialBtnText = t.bankNeed3;

                

                if (p.streakType === 'win') {

                    specialBtnClass += ' btn-bank';

                    specialBtnText = t.bankWinnings;

                } else if (p.streakType === 'lose') {

                    specialBtnClass += ' btn-surrender';

                    specialBtnText = t.surrenderDebt;

                }


                // Ordered Special Buttons

                let specialBtnHtml = `<button class="${specialBtnClass}" onclick="bankOrSurrender(${index})" ${isSpecialEnabled ? '' : 'disabled'}>${specialBtnText}</button>`;

                let halfBtnHtml = `<button class="btn-special btn-half" onclick="halveScore(${index})" ${p.score === 0 ? 'disabled' : ''}>${t.revengeBtn}</button>`;

                let tallyBtnHtml = `<button class="btn-special btn-tally-local" onclick="sweepPlayerToTally(${index})">${t.tallyBtn}</button>`;

                card.innerHTML = `

                        <div class="header-left">

                            <div class="header-name">${p.name}</div>

                        <div class="header-right">

                                ${headerStreakText}

                                ${headerBadgeHtml}

                        </div>

                    </div>

                    

                    <div class="card-body" style="display: ${p.isExpanded ? 'block' : 'none'};">

                        <div class="name-container">

                        ${expandedStreakHtml}

                        <input type="number" id="points-${index}" class="points-input" placeholder="${t.handPoints}" min="0">

                        

                        <div class="action-buttons">

                board.appendChild(card);

            });

                netDisplay = "+" + globalNetScore;

        function updateName(index, newName) {

            saveStateForUndo();

            players[index].name = newName;

            saveData(); 

            render(); 

        }


        function recordHand(index, resultType) {

            saveStateForUndo(); 

            

            const pointsInput = document.getElementById(`points-${index}`);

            const inputPoints = parseInt(pointsInput.value);

            const points = isNaN(inputPoints) ? 0 : inputPoints; 

            const totalHandPoints = points + 5; 


            let p = players[index];


            if (p.streakType === resultType) {

                let currentMagnitude = Math.abs(p.score);

                let newMagnitude = Math.ceil(currentMagnitude * 1.5);

                p.score = p.score >= 0 ? newMagnitude : -newMagnitude;

                p.streakCount++;

            } else {

                p.streakType = resultType;

                p.streakCount = 1;

            }


            if (resultType === 'win') {

                p.score += totalHandPoints;

            } else {

                p.score -= totalHandPoints;

            }


            pointsInput.value = '';

            

            saveData(); 

            render();

        }


        function bankOrSurrender(index) {

            let p = players[index];

            const t = i18n[currentLang];

            let actionText = p.score > 0 ? t.confirmBank : t.confirmSurrender;

            

            if (confirm(`${actionText} ${p.score} ${t.confirmResetStreak}`)) {

                saveStateForUndo(); 

                

                p.setAsidePoints += p.score; 

                p.score = 0;

                p.streakType = null;

                p.streakCount = 0;

                

                saveData(); 

                render();

            }

        }


        function sweepPlayerToTally(index) {

            if (confirm(i18n[currentLang].confirmPlayerPay)) {

                saveStateForUndo(); 

                

                let p = players[index];

                p.runningTally += (p.score + p.setAsidePoints);

                p.score = 0;

                p.streakType = null;

                p.streakCount = 0;

                p.setAsidePoints = 0;

                

                saveData(); 

                render();

            }

        }

                tableOffset = 0;

                players = [

                    { id: 0, name: "Player 1", score: 0, streakType: null, streakCount: 0, setAsidePoints: 0, runningTally: 0, isExpanded: false },

                    { id: 2, name: "Player 3", score: 0, streakType: null, streakCount: 0, setAsidePoints: 0, runningTally: 0, isExpanded: false }

                ];

                

                saveData(); 

                render();

            }

        }
